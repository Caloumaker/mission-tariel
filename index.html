// ============================================================ // MISSION TARIEL — Google Apps Script // ============================================================ // Ce script gère : // 1. L'API pour lire/écrire depuis l'app web // 2. La réinitialisation du marché chaque dimanche soir // 3. Le calcul du classement en fin de semaine // 4. Les notifications via Ntfy.sh // ============================================================ const SHEET_ID = '1a0hnvh8hVwmDHkSU7rcBG5nRXo3XOXaOylgq-arr2ow'; const NTFY_BASE = 'https://ntfy.sh/'; const NTFY_TOPIC_FAMILLE = 'tariel-famille'; // canal commun famille // ============================================================ // UTILS // ============================================================ function getSpreadsheet() { return SpreadsheetApp.openById(SHEET_ID); } function getSheet(name) { return getSpreadsheet().getSheetByName(name); } function getWeekStr(date) { const d = date || new Date(); const jan1 = new Date(d.getFullYear(), 0, 1); const week = Math.ceil(((d - jan1) / (24 * 60 * 60 * 1000) + jan1.getDay() + 1) / 7); return d.getFullYear() + '-W' + String(week).padStart(2, '0'); } function sheetToObjects(sheet) { const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }).filter(r => r[headers[0]]); } function calcPoints(durationMinutes, difficulty) { const dur = parseInt(durationMinutes) || 15; let pts = dur <= 15 ? 1 : dur <= 30 ? 2 : dur <= 60 ? 4 : 5; if (difficulty === 'Moyen') pts += 1; if (difficulty === 'Difficile') pts += 2; return pts; } // ============================================================ // API WEB — point d'entrée HTTP (POST) // ============================================================ function doPost(e) { try { const data = JSON.parse(e.postData.contents); let result; switch(data.action) { case 'assignTask': result = assignTask(data.taskName, data.week, data.userName); break; case 'doneTask': result = doneTask(data.taskName, data.week, data.userName); break; case 'addTask': result = addTask(data.taskName, data.week, data.points, data.category, data.userName); break; case 'getTasks': result = getTasks(data.week); break; case 'getUsers': result = getUsers(); break; default: result = { success: false, error: 'Action inconnue' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // Aussi gérer les requêtes GET pour les lectures function doGet(e) { try { const action = e.parameter.action; const week = e.parameter.week || getWeekStr(); let result; if (action === 'getTasks') result = getTasks(week); else if (action === 'getUsers') result = getUsers(); else result = { success: false, error: 'Action inconnue' }; return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // ============================================================ // ACTIONS SUR LES TACHES // ============================================================ function assignTask(taskName, week, userName) { const sheet = getSheet('Tasks'); const data = sheet.getDataRange().getValues(); const headers = data[0]; const iTask = headers.indexOf('TaskName'); const iWeek = headers.indexOf('Week'); const iAssigned = headers.indexOf('AssignedTo'); const iAssignedDate = headers.indexOf('AssignedDate'); for (let i = 1; i < data.length; i++) { if (data[i][iTask] === taskName && data[i][iWeek] === week) { if (data[i][iAssigned] && data[i][iAssigned] !== '') { return { success: false, error: 'Tâche déjà assignée' }; } sheet.getRange(i + 1, iAssigned + 1).setValue(userName); sheet.getRange(i + 1, iAssignedDate + 1).setValue(new Date().toLocaleDateString('fr-FR')); return { success: true }; } } return { success: false, error: 'Tâche introuvable' }; } function doneTask(taskName, week, userName) { const sheet = getSheet('Tasks'); const data = sheet.getDataRange().getValues(); const headers = data[0]; const iTask = headers.indexOf('TaskName'); const iWeek = headers.indexOf('Week'); const iDone = headers.indexOf('IsDone'); const iDoneBy = headers.indexOf('DoneBy'); const iDoneDate = headers.indexOf('DoneDate'); const iPoints = headers.indexOf('Points'); for (let i = 1; i < data.length; i++) { if (data[i][iTask] === taskName && data[i][iWeek] === week) { sheet.getRange(i + 1, iDone + 1).setValue('TRUE'); sheet.getRange(i + 1, iDoneBy + 1).setValue(userName); sheet.getRange(i + 1, iDoneDate + 1).setValue(new Date().toLocaleDateString('fr-FR')); // Mettre à jour les points du membre dans Users const points = parseInt(data[i][iPoints]) || 0; updateUserPoints(userName, points); // Notifier la famille const msg = `✅ ${userName} vient de réaliser : ${taskName} (+${points} pts) !`; sendNtfyToFamille(msg, 'Mission accomplie !'); return { success: true, points: points }; } } return { success: false, error: 'Tâche introuvable' }; } function addTask(taskName, week, points, category, userName) { const sheet = getSheet('Tasks'); sheet.appendRow([ taskName, week, points, category, userName, new Date().toLocaleDateString('fr-FR'), 'FALSE', '', '', 'TRUE', userName ]); return { success: true }; } function getTasks(week) { const sheet = getSheet('Tasks'); const tasks = sheetToObjects(sheet).filter(t => t.Week === week); return { success: true, tasks: tasks }; } function getUsers() { const sheet = getSheet('Users'); const users = sheetToObjects(sheet); return { success: true, users: users }; } function updateUserPoints(userName, pointsToAdd) { const sheet = getSheet('Users'); const data = sheet.getDataRange().getValues(); const headers = data[0]; const iName = headers.indexOf('Name'); const iPoints = headers.indexOf('Points'); const iTotal = headers.indexOf('PointsTotal'); for (let i = 1; i < data.length; i++) { if (data[i][iName] === userName) { const current = parseInt(data[i][iPoints]) || 0; const total = parseInt(data[i][iTotal]) || 0; sheet.getRange(i + 1, iPoints + 1).setValue(current + pointsToAdd); sheet.getRange(i + 1, iTotal + 1).setValue(total + pointsToAdd); break; } } } // ============================================================ // NOTIFICATIONS NTFY // ============================================================ function sendNtfy(topic, message, title, priority) { try { const url = NTFY_BASE + topic; const options = { method: 'post', headers: { 'Title': title || 'Mission Tariel', 'Priority': priority || 'default', 'Tags': 'house' }, payload: message }; UrlFetchApp.fetch(url, options); } catch(e) { Logger.log('Erreur Ntfy : ' + e.toString()); } } function sendNtfyToFamille(message, title) { sendNtfy(NTFY_TOPIC_FAMILLE, message, title || 'Mission Tariel', 'default'); } function sendNtfyToMembre(userName, message, title) { const sheet = getSheet('Users'); const users = sheetToObjects(sheet); const user = users.find(u => u.Name === userName); if (user && user.NtfyChannel) { sendNtfy(user.NtfyChannel, message, title || 'Mission Tariel', 'default'); } } // ============================================================ // AUTOMATISATION 1 — Réinitialisation du marché (dimanche 20h) // ============================================================ function reinitialiserMarche() { Logger.log('Réinitialisation du marché...'); const week = getWeekStr(); const templateSheet = getSheet('TemplateTasks'); const tasksSheet = getSheet('Tasks'); const templates = sheetToObjects(templateSheet); // Archiver d'abord la semaine écoulée archiverSemaine(); // Supprimer les tâches de la semaine précédente non récurrentes déjà faites nettoyerAnciennesTaches(week); // Vérifier quelles tâches récurrentes existent déjà pour cette semaine const existingTasks = sheetToObjects(tasksSheet) .filter(t => t.Week === week) .map(t => t.TaskName); // Ajouter les tâches récurrentes manquantes let added = 0; templates.forEach(t => { if (t.IsRecurring === 'TRUE' && !existingTasks.includes(t.TaskName)) { const pts = calcPoints(t.DurationMinutes, t.Difficulty); tasksSheet.appendRow([ t.TaskName, week, pts, t.Category, '', '', 'FALSE', '', '', 'FALSE', '' ]); added++; } }); // Vérifier les tâches non assignées et notifier const libres = sheetToObjects(tasksSheet) .filter(t => t.Week === week && (!t.AssignedTo || t.AssignedTo === '')); if (libres.length > 0) { const noms = libres.map(t => `• ${t.TaskName}`).join('\n'); const msg = `🛒 Nouveau marché semaine ${week.split('-W')[1]} !\n${added} tâches disponibles.\n\n${libres.length} tâche(s) libre(s) :\n${noms}\n\nOuvre l'app pour en prendre !`; sendNtfyToFamille(msg, '🏡 Nouveau marché aux missions !'); } Logger.log(`Marché réinitialisé : ${added} tâches ajoutées, ${libres.length} libres.`); } function nettoyerAnciennesTaches(currentWeek) { const sheet = getSheet('Tasks'); const data = sheet.getDataRange().getValues(); const headers = data[0]; const iWeek = headers.indexOf('Week'); const iDone = headers.indexOf('IsDone'); const iExcep = headers.indexOf('IsExceptional'); // Supprimer en partant de la fin pour ne pas décaler les indices for (let i = data.length - 1; i >= 1; i--) { const rowWeek = data[i][iWeek]; const isDone = data[i][iDone]; const isExcep = data[i][iExcep]; // Supprimer tâches exceptionnelles faites des semaines précédentes if (rowWeek !== currentWeek && isExcep === 'TRUE' && isDone === 'TRUE') { sheet.deleteRow(i + 1); } } } // ============================================================ // AUTOMATISATION 2 — Archivage et classement (vendredi 21h) // ============================================================ function archiverSemaine() { const week = getWeekStr(); const tasksSheet = getSheet('Tasks'); const historySheet = getSheet('History'); const usersSheet = getSheet('Users'); const tasks = sheetToObjects(tasksSheet).filter(t => t.Week === week); const users = sheetToObjects(usersSheet); // Calculer le bilan par membre const bilan = users.map(u => { const myTasks = tasks.filter(t => t.AssignedTo === u.Name); const myDone = myTasks.filter(t => t.IsDone === 'TRUE' || t.IsDone === 'true'); const pts = myDone.reduce((s, t) => s + (parseInt(t.Points) || 0), 0); return { name: u.Name, tasksDone: myDone.length, points: pts }; }).sort((a, b) => b.points - a.points); // Attribuer les badges const badges = ['🥇', '🥈', '🥉']; bilan.forEach((b, i) => { if (b.tasksDone === 0) return; const badge = badges[i] || ''; historySheet.appendRow([week, b.name, b.tasksDone, b.points, i + 1, badge]); // Mettre à jour le badge dans Users if (badge) { const uData = usersSheet.getDataRange().getValues(); const uHeaders = uData[0]; const iName = uHeaders.indexOf('Name'); const iBadge = uHeaders.indexOf('Badges'); for (let j = 1; j < uData.length; j++) { if (uData[j][iName] === b.name) { const currentBadges = uData[j][iBadge] || ''; usersSheet.getRange(j + 1, iBadge + 1).setValue(currentBadges + badge); break; } } } }); Logger.log('Archivage semaine ' + week + ' terminé.'); } // ============================================================ // AUTOMATISATION 3 — Rappel du matin (lundi-vendredi 8h) // ============================================================ function rappelMatin() { const week = getWeekStr(); const tasksSheet = getSheet('Tasks'); const usersSheet = getSheet('Users'); const tasks = sheetToObjects(tasksSheet).filter(t => t.Week === week); const users = sheetToObjects(usersSheet); users.forEach(u => { if (!u.NtfyChannel) return; const myTasks = tasks.filter(t => t.AssignedTo === u.Name && (t.IsDone !== 'TRUE' && t.IsDone !== 'true')); if (myTasks.length === 0) return; const noms = myTasks.map(t => `• ${t.TaskName} (${t.Points} pts)`).join('\n'); const msg = `Bonjour ${u.Name} ! 👋\n\nTes tâches restantes cette semaine :\n${noms}\n\nBonne journée !`; sendNtfy(u.NtfyChannel, msg, '🏡 Tes missions du jour', 'default'); }); Logger.log('Rappels du matin envoyés.'); } // ============================================================ // AUTOMATISATION 4 — Alerte tâches libres (mercredi 12h) // ============================================================ function alerteTachesLibres() { const week = getWeekStr(); const tasksSheet = getSheet('Tasks'); const tasks = sheetToObjects(tasksSheet).filter(t => t.Week === week); const libres = tasks.filter(t => !t.AssignedTo || t.AssignedTo === ''); if (libres.length === 0) return; const noms = libres.map(t => `• ${t.TaskName} (${t.Points} pts)`).join('\n'); const msg = `⚠️ ${libres.length} tâche(s) encore libre(s) cette semaine :\n\n${noms}\n\nQui les prend ?`; sendNtfyToFamille(msg, '⚠️ Tâches non assignées'); Logger.log(`Alerte : ${libres.length} tâches libres.`); } // ============================================================ // CLASSEMENT HEBDO — Vendredi 21h // ============================================================ function publierClassement() { const week = getWeekStr(); const tasksSheet = getSheet('Tasks'); const usersSheet = getSheet('Users'); const tasks = sheetToObjects(tasksSheet).filter(t => t.Week === week); const users = sheetToObjects(usersSheet); const bilan = users.map(u => { const myDone = tasks.filter(t => t.AssignedTo === u.Name && (t.IsDone === 'TRUE' || t.IsDone === 'true')); const pts = myDone.reduce((s, t) => s + (parseInt(t.Points) || 0), 0); return { name: u.Name, done: myDone.length, points: pts }; }).sort((a, b) => b.points - a.points).filter(u => u.done > 0); if (bilan.length === 0) return; const medals = ['🥇', '🥈', '🥉']; const lines = bilan.map((u, i) => `${medals[i] || (i+1)+'.'} ${u.name} — ${u.points} pts (${u.done} tâches)`); const msg = `🏆 Classement semaine ${week.split('-W')[1]} :\n\n${lines.join('\n')}\n\nBravo à tous !`; sendNtfyToFamille(msg, '🏆 Classement de la semaine'); Logger.log('Classement publié.'); } // ============================================================ // INSTALLATION DES DÉCLENCHEURS AUTOMATIQUES // ============================================================ function installTriggers() { // Supprimer les anciens déclencheurs ScriptApp.getProjectTriggers().forEach(t => ScriptApp.deleteTrigger(t)); // Dimanche 20h — réinitialisation du marché ScriptApp.newTrigger('reinitialiserMarche') .timeBased().onWeekDay(ScriptApp.WeekDay.SUNDAY).atHour(20).create(); // Lundi au vendredi 8h — rappel du matin [ScriptApp.WeekDay.MONDAY, ScriptApp.WeekDay.TUESDAY, ScriptApp.WeekDay.WEDNESDAY, ScriptApp.WeekDay.THURSDAY, ScriptApp.WeekDay.FRIDAY].forEach(day => { ScriptApp.newTrigger('rappelMatin') .timeBased().onWeekDay(day).atHour(8).create(); }); // Mercredi 12h — alerte tâches libres ScriptApp.newTrigger('alerteTachesLibres') .timeBased().onWeekDay(ScriptApp.WeekDay.WEDNESDAY).atHour(12).create(); // Vendredi 21h — classement hebdo ScriptApp.newTrigger('publierClassement') .timeBased().onWeekDay(ScriptApp.WeekDay.FRIDAY).atHour(21).create(); Logger.log('Déclencheurs installés avec succès !'); } // ============================================================ // TEST — lance manuellement pour vérifier // ============================================================ function testerScript() { Logger.log('=== TEST MISSION TARIEL ==='); Logger.log('Semaine courante : ' + getWeekStr()); const users = getUsers(); Logger.log('Membres : ' + users.users.map(u => u.Name).join(', ')); const tasks = getTasks(getWeekStr()); Logger.log('Tâches semaine : ' + tasks.tasks.length); Logger.log('=== TEST OK ==='); }
